特定数量的同类型元素的集合，内存中的表示一段连续的内存空间； 数组类型定义时，若没有指定其元素值，则每个元素使用类型对应的零值进行初始化；

}

map

map的下标（键）可以是任意的可比较类型

var s []int
s = append(s, 1)

// 以下会报错
var m map[string]string
m["one"] = "first"

// 必须使用在申明后使用 m = make(map[string]string) 创建地址空间

// 嵌套 map
mm = make(map[int]map[int]string)

//错误使用
mm[1][2] = "go"

// 正确使用
if _, ok := mm[1][2]; !ok {
    mm[1] = make(map[int]string)

//map 会自动扩容

//删除 map 中的元素
delete(m,"one")	

//for range 遍历

//map使用结构体为值时：
m["xxx"].name = "jerry" //赋值不被允许，可以取值

// map元素的是无序的

iota 常量

用于定义批量常量时，iota表示基于行的从0开始逐一递增的序列数。

const (
	c1 = iota
	c2 = iota
	c3 = iota
)
fmt.Println(c1,c2,c3) // 0 1 2
	
// 没全部使用 iota 进行定义，iota 还是代表行索引
const (
	c1 = 10
	c2 = 20
	c3 = iota
)
fmt.Println(c1,c2,c3) //10 20 2

// 位运算符
const (
	c1 = 1 << iota
	c2
	c3
)
fmt.Println(c1,c2,c3) // 1 2 4

func httpGet() {
    resp, err := http.Get("http://wwww.baidu.com?id=1")
    if err != nil {
        // handle error
    }
 
    defer resp.Body.Close()

post 请求

func httpPost() {
    resp, err := http.Post("http://www.baidu.com/save",
        "application/x-www-form-urlencoded",
        strings.NewReader("name=tom"))
    if err != nil {
        fmt.Println(err)
    }
 
    defer resp.Body.Close()
    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        // handle error
    }
 
    fmt.Println(string(body))
}

或者：

func httpPostForm() {
    resp, err := http.PostForm("http://www.baidu.com/save",
        url.Values{"name": {"tom"}, "age": {3}})
 
    if err != nil {
        // handle error
    }
 
    defer resp.Body.Close()
    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        // handle error
    }
 
    fmt.Println(string(body))
 
}

通用：

func httpDo() {
    client := &http.Client{}
 
    req, err := http.NewRequest("POST", "http://www.baidu.com/save", strings.NewReader("name=tom"))
    if err != nil {
        // handle error
    }
 
    req.Header.Set("Content-Type", "application/x-www-form-urlencoded")
    req.Header.Set("Cookie", "name=jerry")
 
    resp, err := client.Do(req)
 
    defer resp.Body.Close()
 
    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        // handle error
    }
 
    fmt.Println(string(body))
}
